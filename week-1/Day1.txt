Class Pet
	Attributes Name, Age, Breed
	Constructor(Name, Age, Breed)
		Set this.Name=Name
		Set this.Age=Age
		Set this.Breed=Breed
	Method displayDetail
		SOP(Name)
		SOP(Age)
		SOP(Breed)
