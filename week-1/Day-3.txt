Medical Record Manager
Interface Searchable
    Method search(keyword)

Interface Printable
    Method print()

Class PatientRecord Implements Searchable, Printable
    Attributes PatientID, Name, Diagnosis
    Constructor(PatientID, Name, Diagnosis)
        Set this.PatientID = PatientID
        Set this.Name = Name
        Set this.Diagnosis = Diagnosis
    Method search(keyword)
        IF Name contains keyword OR Diagnosis contains keyword
            Return True
        Else
            Return False
    Method print()
        SOP(PatientID)
        SOP(Name)
        SOP(Diagnosis)

        Set this.ReportID = ReportID

        Set this.Content = Content
    Method print()
        SOP(ReportID)
        SOP(Type)
        SOP(Content)



Patient Record Hashing
Class Patient
    Attributes ID, Name, Age
    Constructor(ID, Name, Age)
        Set this.ID = ID
        Set this.Name = Name
        Set this.Age = Age

Class PatientRecordManager
    Attribute records (HashMap<ID, Patient>)
    Method addPatient(patient)
        records.put(patient.ID, patient)
    Method getPatient(ID)
        Return records.get(ID)
    Method updatePatient(ID, newPatient)
        records.put(ID, newPatient)
    Method deletePatient(ID)
        records.remove(ID)


Maze Solver Bot
Class MazeSolver
    Attributes maze[ ][ ], visited[ ][ ], rows, cols
    Method solve(x, y)
        IF x,y is outside maze OR blocked OR visited
            Return False
        IF x,y is destination
            Return True
        Mark visited[x][y] = True
        IF solve(x+1,y) OR solve(x-1,y) OR solve(x,y+1) OR solve(x,y-1)
            Return True
        Unmark visited[x][y]
        Return False

        Set this.Title = Title
        Set this.Year = Year
        Set this.Rating = Rating
    Method displayBook
        SOP(Title)
        SOP(Year)
        SOP(Rating)

Class Library
    Attribute Books (List<Book>)
    Method filterBooks
        FOR each book in Books
            IF book.Year > 2015 AND book.Rating > 4.5
                SOP(book.Title + " " + book.Rating)
